CREATE TABLE regions (
    region_id NUMBER PRIMARY KEY,
    region_name VARCHAR2(50)
);

CREATE TABLE employees (
    employee_id NUMBER PRIMARY KEY,
    first_name VARCHAR2(50),
    last_name VARCHAR2(50) NOT NULL,
    email VARCHAR2(100) UNIQUE NOT NULL,
    hire_date DATE NOT NULL,
    job_id NUMBER,
    salary NUMBER,
    commission_pct NUMBER(2,2),
    manager_id NUMBER,
    department_id NUMBER
);

CREATE TABLE countries (
    country_id NUMBER PRIMARY KEY,
    country_name VARCHAR2(50),
    region_id NUMBER,
    CONSTRAINT fk_countr_reg FOREIGN KEY (region_id) REFERENCES regions(region_id)
);

CREATE TABLE locations (
    location_id NUMBER PRIMARY KEY,
    street_address VARCHAR2(100),
    postal_code VARCHAR2(20),
    city VARCHAR2(50),
    state_province VARCHAR2(50),
    country_id NUMBER,
    CONSTRAINT fk_loc_countr FOREIGN KEY (country_id) REFERENCES countries(country_id)
);

CREATE TABLE jobs (
    job_id NUMBER PRIMARY KEY,
    job_title VARCHAR2(50) NOT NULL,
    min_salary NUMBER,
    max_salary NUMBER,
    CONSTRAINT check_salary_diff CHECK (max_salary - min_salary >= 2000)
);

CREATE TABLE departments (
    department_id NUMBER PRIMARY KEY,
    department_name VARCHAR2(50) NOT NULL,
    manager_id NUMBER,
    location_id NUMBER,
    CONSTRAINT fk_dept_loc FOREIGN KEY (location_id) REFERENCES locations(location_id)
);

CREATE TABLE job_history (
    employee_id NUMBER,
    start_date DATE,
    end_date DATE,
    job_id NUMBER,
    department_id NUMBER,
    PRIMARY KEY (employee_id, start_date),
    CONSTRAINT fk_jh_emp FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
    CONSTRAINT fk_jh_job FOREIGN KEY (job_id) REFERENCES jobs(job_id),
    CONSTRAINT fk_jh_dept FOREIGN KEY (department_id) REFERENCES departments(department_id)
);
